# 2. VÉRIFICATEUR D'INTÉGRITÉ (SHA256)
# ==============================================================================
class IntegrityVerifier:
    @staticmethod
    def verify_file(filepath: str, expected_hash: str) -> bool:
        if not os.path.exists(filepath):
            return False # Ou True en mode simulation si le fichier n'est pas physiquement présent
        sha256_hash = hashlib.sha256()
        with open(filepath, "rb") as f:
            for byte_block in iter(lambda: f.read(4096), b""):
                sha256_hash.update(byte_block)
        return sha256_hash.hexdigest() == expected_hash

class NeutrinoSpinKeyRotator:
    def __init__(self, base_seed: str = "OBERON_MIND_EX"):
        self.base_seed = base_seed.encode()
        self.rotation_interval_ms = 1
        self._current_key = None
        self._key_timestamp = 0
        
    def _generate_quantum_entropy(self) -> bytes:
        timestamp_ns = time.time_ns()
        cpu_noise = os.urandom(32)
        spin_value = (timestamp_ns ^ int.from_bytes(cpu_noise[:8], 'big')) % (2**64)
        return spin_value.to_bytes(8, 'big') + cpu_noise

    def get_current_key(self) -> bytes:
        current_time_ms = int(time.time() * 1000)
        if (current_time_ms - self._key_timestamp) >= self.rotation_interval_ms:
            entropy = self._generate_quantum_entropy()
            combined = self.base_seed + entropy + str(current_time_ms).encode()
            self._current_key = hashlib.sha3_512(combined).digest()
            self._key_timestamp = current_time_ms
        return self._current_key

    def get_key_id(self) -> str:
        return hashlib.sha256(self.get_current_key()).hexdigest()[:16]

class HMACOberonShield:
    def __init__(self):
        self.key_rotator = NeutrinoSpinKeyRotator()
        self.vaporized_count = 0
        self.validated_count = 0
        self.defense_level = 0
        self.cherenkov_events = []

    def sign_packet(self, payload: bytes, lane: str, metadata: Dict) -> str:
        key = self.key_rotator.get_current_key()
        key_id = self.key_rotator.get_key_id()
        msg_parts = [lane.encode(), str(metadata.get('timestamp', time.time())).encode(), 
                     str(metadata.get('fib_weight', 1)).encode(), payload]
        canonical_msg = b'|'.join(msg_parts)
        signature = hmac.new(key, canonical_msg, hashlib.sha3_512).hexdigest()
        return f"{key_id}:{signature}"

    def verify_packet(self, payload: bytes, lane: str, metadata: Dict, signature: str) -> Tuple[bool, str]:
        try:
            key_id, provided_sig = signature.split(':', 1)
        except ValueError:
            self.vaporized_count += 1
            self._log_cherenkov("INVALID_SIG_FORMAT", lane)
            return False, "Malformed signature format"
        
        key = self.key_rotator.get_current_key()
        for time_offset in [0, -1, -2, 1, 2]: # Tolérance de dérive de 2ms
            adjusted_metadata = metadata.copy()
            adjusted_metadata['timestamp'] = metadata.get('timestamp', time.time()) + (time_offset / 1000)
            msg_parts = [lane.encode(), str(adjusted_metadata['timestamp']).encode(), 
                         str(adjusted_metadata.get('fib_weight', 1)).encode(), payload]
            canonical_msg = b'|'.join(msg_parts)
            expected_sig = hmac.new(key, canonical_msg, hashlib.sha3_512).hexdigest()
            
            if hmac.compare_digest(expected_sig, provided_sig):
                self.validated_count += 1
                return True, "HMAC_VALID"
        
        self.vaporized_count += 1
        self._log_cherenkov("HMAC_MISMATCH", lane)
        self.defense_level = min(self.defense_level + 1, 5)
        return False, f"HMAC_MISMATCH_DEFENSE_LEVEL_{self.defense_level}"

    def _log_cherenkov(self, event_type: str, lane: str):
        self.cherenkov_events.append({
            'timestamp': datetime.utcnow().isoformat(),
            'event_type': event_type,
            'lane': lane,
            'defense_level': self.defense_level
        })
        if len(self.cherenkov_events) > 100:
            self.cherenkov_events = self.cherenkov_events[-100:]

# 5. SECURITY AI (Détection d'Anomalies - Extrait et corrigé de security_ai.py)
# ==============================================================================
class SecurityAI:
    def __init__(self):
        self.alerts: List[Dict] = []
        self.metrics_history: Dict[str, deque] = {
            'tps': deque(maxlen=100),
            'iteration_time': deque(maxlen=100)
        }
        self.baseline_stats: Dict[str, Dict] = {}
        self.anomaly_threshold = 2.5
        self.quant_history = []

    def record_metric(self, metric_name: str, value: float):
        if metric_name in self.metrics_history:
            self.metrics_history[metric_name].append(value)
            if len(self.metrics_history[metric_name]) >= 10:
                self._update_baseline(metric_name)

    def _update_baseline(self, metric_name: str):
        values = list(self.metrics_history[metric_name])
        self.baseline_stats[metric_name] = {
            'mean': statistics.mean(values),
            'stdev': statistics.stdev(values) if len(values) > 1 else 0,
            'updated_at': datetime.now().isoformat()
        }

    def detect_anomaly(self, metric_name: str, value: float) -> Optional[Dict]:
        if metric_name not in self.baseline_stats or self.baseline_stats[metric_name]['stdev'] == 0:
            return None
        baseline = self.baseline_stats[metric_name]
        z_score = abs(value - baseline['mean']) / baseline['stdev']
        if z_score > self.anomaly_threshold:
            alert = {
                'timestamp': datetime.now().isoformat(),
                'type': 'anomaly',
                'metric': metric_name,
                'value': value,
                'z_score': round(z_score, 2),
                'severity': 'high' if z_score > 4 else 'medium'
            }
            self.alerts.append(alert)
            return alert
        return None

    def check_rate_limit(self, requests_per_minute: int, limit: int = 100) -> bool:
        if requests_per_minute > limit:
            self.alerts.append({'type': 'rate_limit_exceeded', 'rpm': requests_per_minute, 'severity': 'high'})
            return False
        return True

    def detect_quantization_attack(self, window: int = 10) -> Optional[Dict]:
        if len(self.quant_history) < window:
            return None
        recent = self.quant_history[-window:]
        most_common = max(set(recent), key=recent.count)
        if recent.count(most_common) / window > 0.8:
            self.alerts.append({'type': 'suspicious_pattern', 'description': f'Repeated requests for {most_common}', 'severity': 'medium'})
            return self.alerts[-1]
        return None

NUM_LAYERS = 12
MATRIX_SIZE = 4
LAYER_RATIO = np.float32(0.625)
INT8_MIN, INT8_MAX = np.int8(-128), np.int8(127)
FIBONACCI_EPOCH = 4272
VRAM_CAPACITY = 4096 # Mode adaptatif via Ring Buffer

@njit(fastmath=True, cache=True)
def numba_forward_pass(input_tensor: np.ndarray, weights: np.ndarray, markov_matrix: np.ndarray) -> np.ndarray:
    current = input_tensor.copy()
    for layer in range(NUM_LAYERS):
        next_tensor = np.zeros((MATRIX_SIZE, MATRIX_SIZE), dtype=np.float32)
        for i in range(MATRIX_SIZE):
            for j in range(MATRIX_SIZE):
                for k in range(MATRIX_SIZE):
                    next_tensor[i, j] += current[i, k] * weights[layer, k, j]
        
        xor_mask = np.float32(((layer * 0x9E3779B9) & 0xFF) / 255.0) 
        scaling_factor = LAYER_RATIO * (1.0 + xor_mask * 0.1)
        markov_influence = markov_matrix[layer % 4, layer % 4]
        next_tensor = next_tensor * scaling_factor * markov_influence
        
        max_abs = np.max(np.abs(next_tensor))
        scale = max_abs / 127.0 if max_abs > 0.0 else 1.0
        
        for i in range(MATRIX_SIZE):
            for j in range(MATRIX_SIZE):
                q_val = np.round(next_tensor[i, j] / scale)
                q_val = np.maximum(INT8_MIN, np.minimum(INT8_MAX, np.int8(q_val)))
                next_tensor[i, j] = np.float32(q_val) * scale
        current = next_tensor
    return current

class PlankQIVRAM_Ring:
    def __init__(self):
        self.buffer = np.zeros((VRAM_CAPACITY, MATRIX_SIZE, MATRIX_SIZE), dtype=np.int8)
        self.head = 0
        np.random.seed(FIBONACCI_EPOCH)
        self.markov = np.abs(np.random.rand(MATRIX_SIZE, MATRIX_SIZE).astype(np.float32))
        self.markov /= self.markov.sum(axis=1, keepdims=True)

    def write(self, data: np.ndarray) -> None:
        self.buffer[self.head] = np.round(data).astype(np.int8)
        self.head = (self.head + 1) % VRAM_CAPACITY
        # Évolution markovienne légère
        noise = np.random.normal(0, 0.002, (MATRIX_SIZE, MATRIX_SIZE)).astype(np.float32)
        self.markov = np.abs(self.markov + noise)
        self.markov /= self.markov.sum(axis=1, keepdims=True)

class QZeroMind:
    def __init__(self):
        self.vram = PlankQIVRAM_Ring()
        np.random.seed(FIBONACCI_EPOCH)
        self.weights = (np.random.rand(NUM_LAYERS, MATRIX_SIZE, MATRIX_SIZE).astype(np.float32)) * LAYER_RATIO
        # Warmup Numba
        _ = numba_forward_pass(np.ones((4,4), dtype=np.float32), self.weights, self.vram.markov)

    def process(self, input_data: np.ndarray) -> Tuple[np.ndarray, float]:
        start_time = time.perf_counter()
        output = numba_forward_pass(input_data, self.weights, self.vram.markov)
        self.vram.write(output)
        return output, time.perf_counter() - start_time

            "status": "PROCESSED",
            "lane": lane,
            "iteration_time_ms": round(iter_time * 1000, 4),
            "output_shape": list(output_tensor.shape),
            "alerts": alerts
        }
